# Sample configuration commands for SNMP

# Basic SNMP configuration
snmp-server community public RO
snmp-server community private RW
snmp-server location DataCenter-1
snmp-server contact admin@example.com

# SNMP Traps
snmp-server enable traps snmp authentication linkdown linkup coldstart warmstart
snmp-server host 192.168.100.10 version 2c public

# ACL for SNMP (optional - be careful with this on production)
access-list 99 permit 192.168.100.0 0.0.0.255
snmp-server community public RO 99
